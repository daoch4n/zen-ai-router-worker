          apiToken: ${{ secrets.CF_API_TOKEN }}
          accountId: ${{ secrets.CF_ACCOUNT_ID }}
          workingDirectory: orchestrator-worker
          command: |
            publish --name router-orchestrator \
            --main src/index.ts \
            --compatibility-date 2024-05-01 \
